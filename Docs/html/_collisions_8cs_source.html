63 public static void ResolveImpulseDynamicADynamicB(Body A, Body B, Vector3 position, Vector3 normal, float delta_time)
64 {
65 float total_mass = A.mass + B.mass;
66 float friction = Mathf.Sqrt(A.friction * B.friction);
67 float elasticity = Mathf.Sqrt(A.elasticity * B.elasticity);
68
69 Vector3 acting_velocity_A = Vector3.Project(A.velocity, normal);
70 Vector3 planar_velocity_A = A.velocity - acting_velocity_A;
71 Vector3 acting_velocity_B = Vector3.Project(B.velocity, normal);
72 Vector3 planar_velocity_B = B.velocity - acting_velocity_B;
73
74 // Get velocity change due to friction
75 Vector3 planar_friction_A = -1.0f * friction * planar_velocity_A;
76 Vector3 planar_friction_B = -1.0f * friction * planar_velocity_B;
77
78 // Get velocity change due to impact reaction
79 // Get momentums in relative to collision
80 Vector3 momentum_A = A.mass * acting_velocity_A;
81 Vector3 momentum_B = B.mass * acting_velocity_B;
82 Vector3 total_momentum = momentum_A + momentum_B;
83
84 /*
85 * (m1 * v1 + m2 * v2 + m2 * e * (v2 - v1)) / (m1 + m2)
86 * (Elasticity * m2 * (v2 - v1) + Total Momentum) / Total Mass
87 */
88 Vector3 reactive_velocity_A = elasticity * B.mass * (acting_velocity_B - acting_velocity_A);
89 reactive_velocity_A += total_momentum;
90 reactive_velocity_A = reactive_velocity_A / total_mass;
91
92 /*
93 * (m1 * v1 + m2 * v2 + m1 * e * (v1 - v2)) / (m1 + m2)
94 * (Elasticity * m1 * (v1 - v2) + Total KE) / Total Mass
95 */
96 Vector3 reactive_velocity_B = elasticity * A.mass * (acting_velocity_A - acting_velocity_B);
97 reactive_velocity_B += total_momentum;
98 reactive_velocity_B = reactive_velocity_B / total_mass;
99
100 // Update velocities: planar friction + reactive
101 Vector3 delta_velocity_A = planar_friction_A + reactive_velocity_A;
102 Vector3 delta_velocity_B = planar_friction_B + reactive_velocity_B;
103 A.AddToAccumulator(Accumulation.Type.velocity, delta_velocity_A);
104 B.AddToAccumulator(Accumulation.Type.velocity, delta_velocity_B);
105
106 // Undo collision in space along normal for most energetic body(s)
107 Vector3 delta_position_A = -1.0f * delta_time * acting_velocity_A;
108 Vector3 delta_position_B = -1.0f * delta_time * acting_velocity_B;
109 if (acting_velocity_A.magnitude > acting_velocity_B.magnitude)
110 {
111 A.AddToAccumulator(Accumulation.Type.position, delta_position_A);
112 }
113 else if (acting_velocity_B.magnitude > acting_velocity_A.magnitude)
114 {
115 B.AddToAccumulator(Accumulation.Type.position, delta_position_B);
116 }
117 else
118 {
119 A.AddToAccumulator(Accumulation.Type.position, delta_position_A);
