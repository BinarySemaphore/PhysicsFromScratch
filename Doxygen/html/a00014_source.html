150 private static float DistanceOverAxis(OBB A, OBB B, Vector3 axis, out int normal, out float center)
151 {
152 normal = 1;
153 center = 0.0f;
154
155 // Find largest and smallest 1d lengths along axis from A and B vertices
156 float min_A = float.MaxValue;
157 float max_A = float.MinValue;
158 float min_B = float.MaxValue;
159 float max_B = float.MinValue;
160
161 for (int i = 0; i < 8; i++)
162 {
163 float vert_A = Vector3.Dot(A.vertices[i], axis);
164 float vert_B = Vector3.Dot(B.vertices[i], axis);
165 if (vert_A < min_A) min_A = vert_A;
166 if (vert_A > max_A) max_A = vert_A;
167 if (vert_B < min_B) min_B = vert_B;
168 if (vert_B > max_B) max_B = vert_B;
169 }
170
171 // Example: |---------------coverage-------------|
172 // |---fill_A----| |--fill_B---|
173 // min_A---------max_A------min_B-------max_B
174 float coverage = Mathf.Max(max_A, max_B) - Mathf.Min(min_A, min_B);
175 float fill_A = max_A - min_A;
176 float fill_B = max_B - min_B;
177 float combined_fill = fill_A + fill_B;
178
179 // Get center
180 float center_A = (max_A - min_A) * 0.5f + min_A;
181 float center_B = (max_B - min_B) * 0.5f + min_B;
182 center = 0.5f * (center_A - center_B);
183
184 // Get normal
185 if (center_A < center_B)
186 {
187 normal = -1;
188 }
189 else
190 {
191 normal = 1;
192 }
193
194 return coverage - combined_fill;
195 }

326 private static bool RayIntersectingQuad(Vector3 start, Vector3 end, Vector3 q1, Vector3 q2, Vector3 q3, Vector3 q4, out Vector3 location, out Vector3 normal)
327 {
328 location = Vector3.zero;
329
330 float line_extension = 0.1f; // extends line limit before and after (zero is exact)
331 float quad_extension = 0.1f; // extends quad range (zero is exact)
332
333 float limit = (end - start).magnitude;
334 Vector3 line_normal = (end - start).normalized;
335 Vector3 plane_normal = Vector3.Cross(q2 - q1, q3 - q1).normalized;
336 normal = plane_normal;
337
338 /*
339 * Line = a1, a2
340 * Plane = q1, q2, q3
341 *
342 * points on the line:
343 * n = normal(a2 - a1)
344 * p = a1 + d * n | where d is distance from a1.
345 *
346 * points on the plane:
347 * N = normal(crossProduct(q2 - q1, q3 - q1))
348 * 0 = dotProduct(P - q1, N)
349 *
350 * Solve for d (distance to plane from a1):
351 * P = p (a point on the line).
352 * 0 = N.x * P.x - N.x * q1.x + N.y * P.y - N.y * q1.y + N.z * P.z - N.z * q1.z
353 * N.x * q1.x + N.y * q1.y + N.z * q1.z = N.x * P.x + N.y * P.y + N.z * P.z
354 * dotProduct(N, q1) = N.x * (a1.x + d * n.x) + N.y * (a1.y + d * n.y) + N.z * (a1.z + d * n.z)
355 * dotProduct(N, q1) = N.x * a1.x + N.x * d * n.x + N.y * a1.y + N.y * d * n.y + N.z * a1.z + N.z * d * n.z
356 * dotProduct(N, q1) - N.x * a1.x - N.y * a1.y - N.z * a1.z = N.x * d * n.x + N.y * d * n.y + N.z * d * n.z
357 * dotProduct(N, q1) - dotProduct(N, a1) = d * (N.x * n.x + N.y * n.y + N.z * n.z)
358 * d = (dotProduct(N, q1) - dotProduct(N, a1)) / dotProduct(N, n)
359 */
360
361 // Check if parallel
362 float direction_compare = Vector3.Dot(line_normal, plane_normal);
363 if (direction_compare == 0f) return false;
364
365 // Get distance to plane from line start
366 float distance = (Vector3.Dot(q1, plane_normal) - Vector3.Dot(start, plane_normal)) / direction_compare;
367
368 // Check if distance is within limit (length of the line from start and end)
369 if (distance < -line_extension || distance > limit + line_extension) return false;
370
371 // Get the point on the plane along the line
372 location = start + distance * line_normal;
373
374 // Check if projection is within quad (sum of anglea from each location to point on edge)
375 float theta = 0.0f;
376 Vector3 check_1, check_2;
377
378 check_1 = q1 - location;
379 check_2 = q2 - location;
380 theta += Mathf.Acos(Vector3.Dot(check_1.normalized, check_2.normalized));
381
382 check_1 = q2 - location;
383 check_2 = q3 - location;
384 theta += Mathf.Acos(Vector3.Dot(check_1.normalized, check_2.normalized));
385
386 check_1 = q3 - location;
387 check_2 = q4 - location;
388 theta += Mathf.Acos(Vector3.Dot(check_1.normalized, check_2.normalized));
389
390 check_1 = q4 - location;
391 check_2 = q1 - location;
392 theta += Mathf.Acos(Vector3.Dot(check_1.normalized, check_2.normalized));
393
394 // Check if sum of all angles (theta) is close to 360 (2pi)
395 if (theta >= 2 * Mathf.PI - quad_extension && theta <= 2 * Mathf.PI + quad_extension) return true;
396
397 return false;
398 }
