131 public static void ResolveImpulseDynamicADynamicB(Body A, Body B, Vector3 position, Vector3 normal, float depth, float delta_time)
132 {
133 float mass_A = A.mass;
134 float mass_B = B.mass;
135 float total_mass = mass_A + mass_B;
136 float friction = Mathf.Sqrt(A.friction * B.friction);
137 float elasticity = Mathf.Sqrt(A.elasticity * B.elasticity);
138
139 Vector3 acting_velocity_A = Vector3.Project(A.velocity, normal);
140 Vector3 planar_velocity_A = A.velocity - acting_velocity_A;
141 Vector3 acting_velocity_B = Vector3.Project(B.velocity, normal);
142 Vector3 planar_velocity_B = B.velocity - acting_velocity_B;
143
144 // Get velocity change due to friction
145 Vector3 planar_friction_A = -1.0f * friction * planar_velocity_A;
146 Vector3 planar_friction_B = -1.0f * friction * planar_velocity_B;
147
148 // Get velocity change due to impact reaction
149 // Get momentums in relative to collision
150 Vector3 momentum_A = mass_A * acting_velocity_A;
151 Vector3 momentum_B = mass_B * acting_velocity_B;
152 Vector3 total_momentum = momentum_A + momentum_B;
153
154 /*
155 * (m1 * v1 + m2 * v2 + m2 * e * (v2 - v1)) / (m1 + m2)
156 * (Elasticity * m2 * (v2 - v1) + Total Momentum) / Total Mass
157 */
158 Vector3 reactive_velocity_A = elasticity * mass_B * (acting_velocity_B - acting_velocity_A);
159 reactive_velocity_A += total_momentum;
160 reactive_velocity_A = reactive_velocity_A / total_mass;
161
162 /*
163 * (m1 * v1 + m2 * v2 + m1 * e * (v1 - v2)) / (m1 + m2)
164 * (Elasticity * m1 * (v1 - v2) + Total KE) / Total Mass
165 */
166 Vector3 reactive_velocity_B = elasticity * mass_A * (acting_velocity_A - acting_velocity_B);
167 reactive_velocity_B += total_momentum;
168 reactive_velocity_B = reactive_velocity_B / total_mass;
169
170 // Update velocities: planar friction + reactive
171 Vector3 delta_velocity_A = planar_friction_A + reactive_velocity_A - 1.1f * acting_velocity_A;
172 Vector3 delta_velocity_B = planar_friction_B + reactive_velocity_B - 1.1f * acting_velocity_B;
173 A.AddToAccumulator(Accumulation.Type.velocity, delta_velocity_A);
174 B.AddToAccumulator(Accumulation.Type.velocity, delta_velocity_B);
175
176 // Resolve collision spacially using depth and percent based on mass or if one object is in a rough spot (high_mass_collision)
177 if (B.awake)
178 {
179 float percent_A = mass_B / total_mass;
180 float percent_B = -1f * mass_A / total_mass;
181 if (A.high_mass_collision > B.high_mass_collision)
182 {
183 percent_A = 0f;
184 percent_B = -1f;
185 }
186 else if (B.high_mass_collision > A.high_mass_collision)

189 percent_B = 0f;
190 }
191 Vector3 delta_position_A = percent_A * 1.1f * depth * normal;
192 Vector3 delta_position_B = percent_B * 1.1f * depth * normal;

194 B.AddToAccumulator(Accumulation.Type.position, delta_position_B);
