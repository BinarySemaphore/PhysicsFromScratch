114 public static void ResolveImpulseDynamicAGroundB(Body A, Vector3 position, Vector3 normal, float depth, float delta_time)
115 {
116 float friction = A.friction;
117 float elasticity = A.elasticity;
118
119 Vector3 position_to_center = A.transform.position - position;
120 float distance_to_center = position_to_center.magnitude;
121
122 Vector3 velocity_A_at_pos = A.velocity;
123
124 // Add velocity from rotation relative to position of collision
125 Vector3 velocity_from_rotation = Vector3.Cross(A.transform.rotation * A.r_velocity, -position_to_center);
126 velocity_from_rotation *= distance_to_center * delta_time;
127 velocity_A_at_pos += velocity_from_rotation;
128
129 Vector3 acting_velocity = Vector3.Project(velocity_A_at_pos, normal);
130 Vector3 planar_velocity = velocity_A_at_pos - acting_velocity;
131
132 // Get velocity change due to friction
133 Vector3 planar_friction = -1.0f * friction * planar_velocity;
134
135 // Get velocity change due to impact reaction
136 Vector3 reactive_velocity = -1.0f * elasticity * acting_velocity;
137
138 // Update velocities: planar friction + reactive - acting
139 Vector3 delta_velocity = planar_friction + reactive_velocity - acting_velocity;
140 A.AddToAccumulator(Accumulation.Type.velocity, delta_velocity);
141
142 // Update rotational velocities
143 Vector3 torque = Vector3.Cross(delta_velocity, position_to_center);
144 torque = Quaternion.Inverse(A.transform.rotation) * torque;
145 Vector3 delta_r_velocity = Vector3.Scale(torque, new Vector3(distance_to_center / A.moment.x, distance_to_center / A.moment.y, distance_to_center / A.moment.z));
146 A.AddToAccumulator(Accumulation.Type.r_velocity, delta_r_velocity);
147
148 // Undo collision in space along normal
149 Vector3 delta_position = new Vector3(0f, depth, 0f);
150 //A.transform.position += delta_position;

163 public static void ResolveImpulseDynamicADynamicB(Body A, Body B, Vector3 position, Vector3 normal, float depth, float delta_time)
164 {
165 float mass_A = A.mass;
166 float mass_B = B.mass;
167 float total_mass = mass_A + mass_B;
168 float friction = Mathf.Sqrt(A.friction * B.friction);
169 float elasticity = Mathf.Sqrt(A.elasticity * B.elasticity);
170
171 Vector3 position_to_A = A.transform.position - position;
172 Vector3 position_to_B = B.transform.position - position;
173 float distance_to_A = position_to_A.magnitude;
174 float distance_to_B = position_to_B.magnitude;
175
176 Vector3 velocity_A_at_pos = A.velocity;
177 Vector3 velocity_B_at_pos = B.velocity;
178
179 // Add velocity from rotation relative to position of collision
180 Vector3 velocity_from_rotation_A = Vector3.Cross(A.transform.rotation * A.r_velocity, -position_to_A);
181 velocity_from_rotation_A *= distance_to_A * delta_time;
182 Vector3 velocity_from_rotation_B = Vector3.Cross(A.transform.rotation * A.r_velocity, -position_to_B);
183 velocity_from_rotation_B *= distance_to_B * delta_time;
184 velocity_A_at_pos += velocity_from_rotation_A;
185 velocity_B_at_pos += velocity_from_rotation_B;
186
187 // Get velocities contributing and perpendicular to collision
188 Vector3 acting_velocity_A = Vector3.Project(velocity_A_at_pos, normal);
189 Vector3 planar_velocity_A = velocity_A_at_pos - acting_velocity_A;
190 Vector3 acting_velocity_B = Vector3.Project(velocity_B_at_pos, normal);
191 Vector3 planar_velocity_B = velocity_B_at_pos - acting_velocity_B;
192
193 // Get velocity change due to friction
194 Vector3 planar_friction_A = -1.0f * friction * planar_velocity_A;
195 Vector3 planar_friction_B = -1.0f * friction * planar_velocity_B;
196
197 // Get velocity change due to impact reaction
198 // Get momentums in relative to collision
199 Vector3 momentum_A = mass_A * acting_velocity_A;
200 Vector3 momentum_B = mass_B * acting_velocity_B;
201 Vector3 total_momentum = momentum_A + momentum_B;
202
203 /*
204 * (m1 * v1 + m2 * v2 + m2 * e * (v2 - v1)) / (m1 + m2)
205 * (Elasticity * m2 * (v2 - v1) + Total Momentum) / Total Mass
206 */
207 Vector3 reactive_velocity_A = elasticity * mass_B * (acting_velocity_B - acting_velocity_A);
208 reactive_velocity_A += total_momentum;
209 reactive_velocity_A = reactive_velocity_A / total_mass;
210
211 /*
212 * (m1 * v1 + m2 * v2 + m1 * e * (v1 - v2)) / (m1 + m2)
213 * (Elasticity * m1 * (v1 - v2) + Total Momentum) / Total Mass
214 */
215 Vector3 reactive_velocity_B = elasticity * mass_A * (acting_velocity_A - acting_velocity_B);
216 reactive_velocity_B += total_momentum;
217 reactive_velocity_B = reactive_velocity_B / total_mass;
218
219 // Update velocities: planar friction + reactive
220 Vector3 delta_velocity_A = planar_friction_A + reactive_velocity_A - 1.1f * acting_velocity_A;
221 Vector3 delta_velocity_B = planar_friction_B + reactive_velocity_B - 1.1f * acting_velocity_B;
222 A.AddToAccumulator(Accumulation.Type.velocity, delta_velocity_A);
223 B.AddToAccumulator(Accumulation.Type.velocity, delta_velocity_B);
224
225 // Update rotational velocities
226 Vector3 torque_A = Vector3.Cross(delta_velocity_A, position_to_A);
227 torque_A = Quaternion.Inverse(A.transform.rotation) * torque_A;
228 Vector3 torque_B = Vector3.Cross(delta_velocity_B, position_to_B);
229 torque_B = Quaternion.Inverse(B.transform.rotation) * torque_B;
230 Vector3 delta_r_velocity_A = Vector3.Scale(torque_A, new Vector3(distance_to_A / A.moment.x, distance_to_A / A.moment.y, distance_to_A / A.moment.z));
231 Vector3 delta_r_velocity_B = Vector3.Scale(torque_B, new Vector3(distance_to_B / B.moment.x, distance_to_B / B.moment.y, distance_to_B / B.moment.z));
232 A.AddToAccumulator(Accumulation.Type.r_velocity, delta_r_velocity_A);
233 B.AddToAccumulator(Accumulation.Type.r_velocity, delta_r_velocity_B);
234
235 // Resolve collision spacially using depth and percent based on mass or if one object is in a rough spot (high_mass_collision)
236 if (B.awake)
237 {
238 float percent_A = mass_B / total_mass;
239 float percent_B = -1f * mass_A / total_mass;
240 if (A.high_mass_collision > B.high_mass_collision)
241 {
242 percent_A = 0f;
243 percent_B = -1f;
244 }
245 else if (B.high_mass_collision > A.high_mass_collision)
246 {
247 percent_A = 1f;
248 percent_B = 0f;
249 }
250 Vector3 delta_position_A = percent_A * 1.1f * depth * normal;
251 Vector3 delta_position_B = percent_B * 1.1f * depth * normal;
252 A.AddToAccumulator(Accumulation.Type.position, delta_position_A);
253 B.AddToAccumulator(Accumulation.Type.position, delta_position_B);
254 }
255 else
256 {
257 Vector3 delta_position_A = depth * normal;
258 A.AddToAccumulator(Accumulation.Type.position, delta_position_A);
259 if (A.high_mass_collision < 1) A.high_mass_collision = 10;
260 }
261
262 // Transfer high mass collision
263 if (A.high_mass_collision > B.high_mass_collision) B.high_mass_collision = A.high_mass_collision - 1;
264 if (B.high_mass_collision > A.high_mass_collision) A.high_mass_collision = B.high_mass_collision - 1;
265 }
