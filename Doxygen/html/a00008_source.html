142 public static void ResolveImpulseDynamicADynamicB(Body A, Body B, Vector3 position, Vector3 normal, float depth, float delta_time)
143 {
144 float mass_A = A.mass;
145 float mass_B = B.mass;
146 float total_mass = mass_A + mass_B;
147 float friction = Mathf.Sqrt(A.friction * B.friction);
148 float elasticity = Mathf.Sqrt(A.elasticity * B.elasticity);
149
150 Vector3 acting_velocity_A = Vector3.Project(A.velocity, normal);
151 Vector3 planar_velocity_A = A.velocity - acting_velocity_A;
152 Vector3 acting_velocity_B = Vector3.Project(B.velocity, normal);
153 Vector3 planar_velocity_B = B.velocity - acting_velocity_B;
154
155 // Get velocity change due to friction
156 Vector3 planar_friction_A = -1.0f * friction * planar_velocity_A;
157 Vector3 planar_friction_B = -1.0f * friction * planar_velocity_B;
158
159 // Get velocity change due to impact reaction
160 // Get momentums in relative to collision
161 Vector3 momentum_A = mass_A * acting_velocity_A;
162 Vector3 momentum_B = mass_B * acting_velocity_B;
163 Vector3 total_momentum = momentum_A + momentum_B;
164
165 /*
166 * (m1 * v1 + m2 * v2 + m2 * e * (v2 - v1)) / (m1 + m2)
167 * (Elasticity * m2 * (v2 - v1) + Total Momentum) / Total Mass
168 */
169 Vector3 reactive_velocity_A = elasticity * mass_B * (acting_velocity_B - acting_velocity_A);
170 reactive_velocity_A += total_momentum;
171 reactive_velocity_A = reactive_velocity_A / total_mass;
172
173 /*
174 * (m1 * v1 + m2 * v2 + m1 * e * (v1 - v2)) / (m1 + m2)
175 * (Elasticity * m1 * (v1 - v2) + Total KE) / Total Mass
176 */
177 Vector3 reactive_velocity_B = elasticity * mass_A * (acting_velocity_A - acting_velocity_B);
178 reactive_velocity_B += total_momentum;
179 reactive_velocity_B = reactive_velocity_B / total_mass;
180
181 // Update velocities: planar friction + reactive
182 Vector3 delta_velocity_A = planar_friction_A + reactive_velocity_A - 1.1f * acting_velocity_A;
183 Vector3 delta_velocity_B = planar_friction_B + reactive_velocity_B - 1.1f * acting_velocity_B;
184 A.AddToAccumulator(Accumulation.Type.velocity, delta_velocity_A);
185 B.AddToAccumulator(Accumulation.Type.velocity, delta_velocity_B);
186
187 // Resolve collision spacially using depth and percent based on mass or if one object is in a rough spot (high_mass_collision)
188 if (B.awake)
189 {
190 float percent_A = mass_B / total_mass;
191 float percent_B = -1f * mass_A / total_mass;
192 if (A.high_mass_collision > B.high_mass_collision)
193 {
194 percent_A = 0f;
195 percent_B = -1f;
196 }
197 else if (B.high_mass_collision > A.high_mass_collision)
198 {
199 percent_A = 1f;
200 percent_B = 0f;
201 }
202 Vector3 delta_position_A = percent_A * 1.1f * depth * normal;
203 Vector3 delta_position_B = percent_B * 1.1f * depth * normal;
204 A.AddToAccumulator(Accumulation.Type.position, delta_position_A);
205 B.AddToAccumulator(Accumulation.Type.position, delta_position_B);
206 }
207 else
208 {
209 Vector3 delta_position_A = depth * normal;
210 A.AddToAccumulator(Accumulation.Type.position, delta_position_A);
211 if (A.high_mass_collision < 1) A.high_mass_collision = 10;
212 }
213
214 // Transfer high mass collision
215 if (A.high_mass_collision > B.high_mass_collision) B.high_mass_collision = A.high_mass_collision - 1;
216 if (B.high_mass_collision > A.high_mass_collision) A.high_mass_collision = B.high_mass_collision - 1;
217 }
