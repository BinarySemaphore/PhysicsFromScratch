149 private List<Collision> GetCollisions(float delta_time)
150 {
151 List<Collision> collisions = new List<Collision>();
152
153 // Get neighbors from subdivisions in octree
154 List<Body> neighbors = new List<Body>();
155 foreach (Octree subdivision in this.bounding_box.subdivisions[this.bounding_box.subdivisions.Count - 1])
156 {
157 foreach (OctreeItem neighbor in subdivision.items)
158 {
159 if (neighbor.body != this) neighbors.Add(neighbor.body);
160 }
161 // Remove self from subdivision items to prevent rechecking same pairs
162 subdivision.items.Remove(this.bounding_box);
163 }
164
165 // Find collisions
166 collisions.AddRange(Collision.GetCollisionsForDynamicNeighbors(this, neighbors));
167 collisions.AddRange(Collision.GetCollisionsForGround(this, delta_time));
168
169 return collisions;
