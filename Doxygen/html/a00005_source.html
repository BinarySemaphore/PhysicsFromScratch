141 private List<Collision> GetCollisions(float delta_time)
142 {
143 List<Collision> collisions = new List<Collision>();
144
145 // Get neighbors from subdivisions in octree
146 List<Body> neighbors = new List<Body>();
147 foreach (Octree subdivision in this.bounding_box.subdivisions[this.bounding_box.subdivisions.Count - 1])
148 {
149 foreach (OctreeItem neighbor in subdivision.items)
150 {
151 if (neighbor.body != this) neighbors.Add(neighbor.body);
152 }
153 // Remove self from subdivision items to prevent rechecking same pairs
154 subdivision.items.Remove(this.bounding_box);
155 }
156
157 // Find collisions
158 collisions.AddRange(Collision.GetCollisionsForDynamicNeighbors(this, neighbors));
159 collisions.AddRange(Collision.GetCollisionsForGround(this, delta_time));
